CREATE TABLE student (Student_ID INT AUTO_INCREMENT PRIMARY KEY,
Name VARCHAR(255) NOT NULL,
Department VARCHAR(255) NOT NULL,
Course VARCHAR(255) NOT NULL,
Year VARCHAR(255) NOT NULL,
Semester VARCHAR(255) NOT NULL,
Division VARCHAR(255) NOT NULL,
Gender VARCHAR(255) NOT NULL,
DOB DATE NOT NULL,
Mobile_No VARCHAR(255) NOT NULL,
Address VARCHAR(255) NOT NULL,
Roll_No VARCHAR(255) NOT NULL,
Email VARCHAR(255) NOT NULL,
Teacher_Name VARCHAR(255) NOT NULL,
PhotoSample VARCHAR(255) NOT NULL
);


CREATE TABLE regteach (
id INT AUTO_INCREMENT PRIMARY KEY,
first_name VARCHAR(255) NOT NULL,
last_name VARCHAR(255) NOT NULL,
contact_number VARCHAR(15) NOT NULL,
email VARCHAR(255) NOT NULL,
security_question VARCHAR(255) NOT NULL,
security_answer VARCHAR(255) NOT NULL,
password VARCHAR(255) NOT NULL
);

CREATE TABLE stdattendance (
std_id INT AUTO_INCREMENT PRIMARY KEY,
std_roll_no VARCHAR(20),
std_name VARCHAR(255),
std_time VARCHAR(20),
std_date VARCHAR(20),
std_attendance VARCHAR(20)
);
